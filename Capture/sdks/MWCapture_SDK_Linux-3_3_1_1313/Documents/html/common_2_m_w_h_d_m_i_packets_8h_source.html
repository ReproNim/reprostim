333 typedef struct _HDMI_INFOFRAME_PACKET {
334  HDMI_INFOFRAME_HEADER header;
335  uint8_t byChecksum;
336 
337  union {
338  uint8_t abyPayload[27];
339  HDMI_AVI_INFOFRAME_PAYLOAD aviInfoFramePayload;
340  HDMI_AUDIO_INFOFRAME_PAYLOAD audioInfoFramePayload;
341  HDMI_SPD_INFOFRAME_PAYLOAD spdInfoFramePayload;
342  HDMI_VS_INFOFRAME_PAYLOAD vsInfoFramePayload;
343  };
344 
345 public:
346  bool IsValid(
347  ) const {
348  uint8_t * pbyData = (uint8_t *)&header;
349  uint8_t cbData = header.byLength + sizeof(header) + 1;
350 
351  uint8_t bySum = 0;
352  while (cbData-- != 0)
353  bySum += *pbyData++;
354 
355  return (bySum == 0);
356  }
357 } HDMI_INFOFRAME_PACKET;
