375 static inline BOOLEAN hdmi_info_IsValid(HDMI_INFOFRAME_PACKET *info)
376 {
377  unsigned char *pbyData = (unsigned char *)&info->header;
378  BYTE cbData = info->header.byLength + sizeof(info->header) + 1;
379 
380  BYTE bySum = 0;
381  while (cbData-- != 0)
382  bySum += *pbyData++;
383 
384  return (bySum == 0);
385 }
